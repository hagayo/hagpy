async def get_dream(dream_id: UUID) -> Dream:
    key = f"dream:{dream_id}"
    cached = await redis.get(key)
    if cached is not None:
        return Dream.model_validate_json(cached)

    dream = await repository.get(dream_id)
    await redis.set(key, dream.model_dump_json(), ex=60)
    return dream
