services/dream_service.py
class DreamService:
    def __init__(self, repository: DreamRepository) -> None:
        self._repository = repository

    def create(self, command: CreateDream) -> Dream:
        if self._repository.title_exists(command.title):
            raise DreamAlreadyExists(command.title)

        dream = Dream.create(command.title, command.description)
        self._repository.save(dream)
        return dream
