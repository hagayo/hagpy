class InMemoryDreamRepository:
    def __init__(self) -> None:
        self._items: dict[UUID, Dream] = {}

    def save(self, dream: Dream) -> None:
        self._items[dream.id] = dream

    def get(self, dream_id: UUID) -> Dream | None:
        return self._items.get(dream_id)
